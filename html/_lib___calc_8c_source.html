144Std_ReturnType Lib_Calc_Div(sint32 arg1, sint32 arg2, sint32* result)
145{
146 Std_ReturnType ret = E_OK;
147 sint64 tmp = 0;
148
149 if (arg2 == 0L) {
150 if (arg1 > 0L) {
151 tmp = DIV_POS_BY_ZERO_RESULT;
152 } else if (arg1 < 0L) {
153 tmp = DIV_NEG_BY_ZERO_RESULT;
154 } else {
155 tmp = DIV_ZERO_BY_ZERO_RESULT;
156 }
157 ret = E_NOT_OK;
158
159 } else {
160 tmp = arg1 / arg2;
161 if ((tmp > MAX32) || (tmp < MIN32)) {
162 ret = E_NOT_OK;
163 }
164 }
165 *result = (sint32)tmp;
166 return ret;
167}
